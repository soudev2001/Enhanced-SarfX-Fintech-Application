        // --- FETCH STATS (API + FALLBACK) ---
        async function fetchStats() {
            // Default random fallback
            const randomFallback = Math.floor(Math.random() * (500 - 150 + 1)) + 150;
            let data = { real_count: randomFallback, count: randomFallback, socials: {} };

            try {
                const res = await fetch('/api/public-data');
                if (res.ok) {
                    const realData = await res.json();
                    if (realData.real_count !== undefined && realData.real_count !== null) {
                         data.real_count = realData.real_count;
                    }
                    if (realData.count !== undefined && realData.count !== null) {
                         data.count = realData.count;
                    }
                    if (realData.socials) data.socials = realData.socials;
                }
            } catch(e) { } 
            
            const counter = document.getElementById('counter');
            const end = data.real_count; // Utiliser le nombre r√©el au lieu du total
            document.getElementById('user-rank').innerText = end + 1;
            
            let current = 0;
            const step = Math.ceil(end / 40) || 1; 
            const timer = setInterval(() => {
                current += step;
                if (current >= end) { current = end; clearInterval(timer); }
                counter.innerText = "+" + current;
                const mini = document.getElementById('mini-counter');
                if(mini) mini.innerText = current;
            }, 30);

            const soc = document.getElementById('social-links');
            if (soc) {
                soc.innerHTML = '';
                const s = data.socials;
                if(s.instagram) soc.innerHTML += `<a href="${s.instagram}" target="_blank" class="hover:text-[#e05a03] transition"><i data-lucide="instagram"></i></a>`;
                if(s.linkedin) soc.innerHTML += `<a href="${s.linkedin}" target="_blank" class="hover:text-[#e05a03] transition"><i data-lucide="linkedin"></i></a>`;
                if(s.twitter) soc.innerHTML += `<a href="${s.twitter}" target="_blank" class="hover:text-[#e05a03] transition"><i data-lucide="twitter"></i></a>`;
                lucide.createIcons();
            }
        }
